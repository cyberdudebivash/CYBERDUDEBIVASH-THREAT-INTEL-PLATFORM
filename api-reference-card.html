Python SDK Pattern
PYTHON — Production Client 
"""
CYBERDUDEBIVASH SENTINEL APEX — Python API Client
Production-grade: session pooling, retry, rate-limit handling
"""
import requests, time, logging
from typing import Optional, Dict, Any

logging.basicConfig(level=logging.INFO)
log = logging.getLogger("sentinel-apex")

BASE_URL = "https://intel.cyberdudebivash.com"

class SentinelApexClient:
    def __init__(self, api_key: str, timeout: int = 15):
        self.api_key = api_key
        self.timeout = timeout
        self.session = requests.Session()
        self.session.headers.update({
            "X-API-Key": api_key,
            "Accept": "application/json",
            "User-Agent": "SentinelApex-PythonClient/1.0"
        })
        self._jwt: Optional[str] = None

    def _request(self, method: str, path: str, **kwargs) -> Dict[str, Any]:
        url = f"{BASE_URL}{path}"
        for attempt in range(3):
            try:
                r = self.session.request(method, url, timeout=self.timeout, **kwargs)
                if r.status_code == 429:
                    retry_after = int(r.headers.get("Retry-After", 60))
                    log.warning(f"Rate limited. Sleeping {retry_after}s")
                    time.sleep(retry_after); continue
                r.raise_for_status()
                return r.json()
            except requests.RequestException as e:
                if attempt == 2: raise
                time.sleep(2 ** attempt)
        raise RuntimeError("Max retries exceeded")

    def health(self): return self._request("GET", "/api/health")
    def feed(self): return self._request("GET", "/api/feed.json")
    def latest(self): return self._request("GET", "/api/v1/intel/latest.json")
    def top10(self): return self._request("GET", "/api/v1/intel/top10.json")
    def apex(self): return self._request("GET", "/api/v1/intel/apex.json")

    def search(self, query: str):
        return self._request("GET", "/api/v1/intel/search", params={"q": query})

    def get_feed(self, page: int = 1, limit: int = 50, severity: str = None):
        params = {"page": page, "limit": limit}
        if severity: params["severity"] = severity
        return self._request("GET", "/api/v1/intel/feed", params=params)

    def get_stix(self, stix_id: str):
        return self._request("GET", f"/api/v1/stix/{stix_id}")

    def iocs(self, ioc_type: str = None, limit: int = 100):
        params = {"limit": limit}
        if ioc_type: params["type"] = ioc_type
        return self._request("GET", "/api/v1/iocs", params=params)

// CYBERDUDEBIVASH SENTINEL APEX — JS/Node Client
// Works in browser, Node.js 18+, Deno, Bun, CF Workers

const BASE = 'https://intel.cyberdudebivash.com';
const API_KEY = 'YOUR_API_KEY_HERE';

const apex = {
  headers: () => ({
    'X-API-Key': API_KEY,
    'Accept': 'application/json'
  }),

  async get(path, params = {}) {
    const url = new URL(`${BASE}${path}`);
    Object.entries(params).forEach(([k, v]) => url.searchParams.set(k, v));
    const res = await fetch(url, { headers: this.headers() });
    if (!res.ok) throw new Error(`API ${res.status}: ${res.statusText}`);
    return res.json();
  },

  health:   () => apex.get('/api/health'),
  feed:     () => apex.get('/api/feed.json'),
  latest:   () => apex.get('/api/v1/intel/latest.json'),
  top10:    () => apex.get('/api/v1/intel/top10.json'),
  apex:     () => apex.get('/api/v1/intel/apex.json'),
  search:   (q) => apex.get('/api/v1/intel/search', { q }),
  getIntel: (id) => apex.get(`/api/v1/intel/${id}`),
  iocs:     (type) => apex.get('/api/v1/iocs', type ? { type } : {}),
  stix:     (id) => apex.get(`/api/v1/stix/${id}`),
};

// Usage
const { version, advisory_count } = await apex.health();
console.log(`SENTINEL APEX v${version} — ${advisory_count} advisories live`);

Go Client
GO — Production HTTP Client 
package main

import (
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "time"
)

const (
    BaseURL = "https://intel.cyberdudebivash.com"
    APIKey  = "YOUR_API_KEY_HERE"
)

type Client struct {
    http   *http.Client
    apiKey string
}

func NewClient(apiKey string) *Client {
    return &Client{
        http:   &http.Client{Timeout: 15 * time.Second},
        apiKey: apiKey,
    }
}

func (c *Client) Get(path string) (map[string]interface{}, error) {
    req, _ := http.NewRequest("GET", BaseURL+path, nil)
    req.Header.Set("X-API-Key", c.apiKey)
    req.Header.Set("Accept", "application/json")
    resp, err := c.http.Do(req)
    if err != nil { return nil, err }
    defer resp.Body.Close()
    body, _ := io.ReadAll(resp.Body)
    var result map[string]interface{}
    return result, json.Unmarshal(body, &result)
}

func main() {
    client := NewClient(APIKey)
    health, err := client.Get("/api/health")
    if err != nil { panic(err) }
    fmt.Printf("SENTINEL APEX v%v — %v advisories\n",
        health["version"], health["advisory_count"])
}

Security: Never expose API keys in client-side JavaScript, public repos, or logs. Use environment variables. Rotate keys immediately if compromised via the Admin portal.

QUICK START

Integrate in under 5 minutes with any language:

# Python
import requests
data = requests.get("https://intel.cyberdudebivash.com/api/feed.json").json()
critical = [d for d in data["data"] if d["risk_score"] >= 9]
print(f"{len(critical)} critical threats found")
// JavaScript
const res = await fetch("https://intel.cyberdudebivash.com/api/feed.json");
const { data } = await res.json();
const kev = data.filter(d => d.kev_present);
console.log(`${kev.length} CISA KEV items`);
